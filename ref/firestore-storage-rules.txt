rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    function signedIn() { return request.auth != null; }
    function isAdmin() {
      return signedIn() &&
        firestore.exists(/databases/(default)/documents/admins/$(request.auth.uid));
    }

    // We store at: orders/{orderId}/{filename}.jpg
    match /orders/{orderId}/{fileName} {
      allow read: if true;          // simplest: token protects the link, not the image URL
      allow write: if signedIn();   // anonymous OK
    }
  }
}
